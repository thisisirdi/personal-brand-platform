Data Model
projects
model Project {
  id          String   @id @default(uuid())
  title       String
  slug        String   @unique
  summary     String
  content     String
  category    String
  techStack   String[]
  status      String   @default("published")
  featured    Boolean  @default(false)
  orderIndex  Int      @default(0)
  githubUrl   String?
  demoUrl     String?
  imageUrl    String?
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt
}
contacts
model Contact {
  id        String   @id @default(uuid())
  name      String
  email     String
  message   String
  source    String? 
  createdAt DateTime @default(now())
}
